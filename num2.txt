SELECT
CONCAT(a.fName, ' ', a.mName, ' ', a.lName) AS fullname, b.title,
DATE_FORMAT(c.requestedDate, '%M') AS month, YEAR(c.requestedDate) AS year,
SUM(c.requestedQty) AS requestedQty FROM tbl_customer_info a JOIN
tbl_rents_info c ON a.id = c.tbl_customer_info_id JOIN tbl_movie_info b ON b.id
= c.tbl_movie_info_id GROUP BY fullname, b.title, month, year ORDER BY
fullname, year, STR_TO_DATE(month, '%M'), b.title;
